## Compound Components（複合元件）

讓元件共享隱式狀態，使用更直覺。

```tsx
// Select 複合元件
const SelectContext = createContext<SelectContextType>(null!);

function Select({ children, value, onChange }) {
  return (
    <SelectContext.Provider value={{ value, onChange }}>
      <div className="select">{children}</div>
    </SelectContext.Provider>
  );
}

function Option({ value, children }) {
  const { value: selected, onChange } = useContext(SelectContext);
  return (
    <div
      className={`option ${selected === value ? 'active' : ''}`}
      onClick={() => onChange(value)}
    >
      {children}
    </div>
  );
}

Select.Option = Option;

// 使用
<Select value={role} onChange={setRole}>
  <Select.Option value="admin">管理員</Select.Option>
  <Select.Option value="user">一般用戶</Select.Option>
</Select>
```

## Render Props

```tsx
// 把渲染邏輯傳出去
function DataFetcher<T>({ url, render }: {
  url: string;
  render: (data: T | null, loading: boolean) => ReactNode;
}) {
  const { data, loading } = useFetch<T>(url);
  return <>{render(data, loading)}</>;
}

// 使用
<DataFetcher
  url="/api/users"
  render={(users, loading) =>
    loading ? <Spinner /> : <UserList users={users} />
  }
/>
```

## HOC（Higher-Order Component）

```tsx
// 包裝元件，注入額外功能
function withAuth<P>(Component: ComponentType<P>) {
  return function AuthGuard(props: P) {
    const { user, loading } = useAuth();

    if (loading) return <PageSpinner />;
    if (!user) return <Navigate to="/login" />;

    return <Component {...props} />;
  };
}

// 使用
const ProtectedDashboard = withAuth(Dashboard);
```

## Custom Hooks（最佳實踐）

> 現代 React 中，大部分 Render Props 和 HOC 都應用 Custom Hooks 取代。

```tsx
// 比 HOC 更簡潔
function Dashboard() {
  const { user, loading } = useAuth();
  if (loading) return <Spinner />;
  if (!user) return <Navigate to="/login" />;
  return <DashboardContent user={user} />;
}
```

## Controlled vs Uncontrolled

```tsx
// Controlled — React 控制值
function ControlledInput() {
  const [value, setValue] = useState('');
  return <input value={value} onChange={e => setValue(e.target.value)} />;
}

// Uncontrolled — DOM 自行管理
function UncontrolledForm() {
  const inputRef = useRef<HTMLInputElement>(null);

  const handleSubmit = () => {
    console.log(inputRef.current?.value);
  };

  return <input ref={inputRef} defaultValue="初始值" />;
}
```

## Container / Presenter 模式

```tsx
// Container：管理邏輯與資料
function UserListContainer() {
  const { data: users, loading } = useFetch('/api/users');
  const handleDelete = useCallback(async (id) => {
    await api.delete(`/users/${id}`);
  }, []);

  return <UserListView users={users} loading={loading} onDelete={handleDelete} />;
}

// Presenter：只負責顯示
function UserListView({ users, loading, onDelete }) {
  if (loading) return <Skeleton />;
  return (
    <ul>
      {users?.map(u => (
        <li key={u.id}>
          {u.name}
          <button onClick={() => onDelete(u.id)}>刪除</button>
        </li>
      ))}
    </ul>
  );
}
```
